\section{Probability}
\begin{framed}
  \textbf{Chain Rule:}
  \begin{equation}
    P \left( \bigcap_{k=1}^n \right) = \prod_{k=1}^n P \left( A_k \mid \bigcap_{j=1}^{k-1} A_j \right)
  \end{equation}
  For $n=2$:
  \begin{equation}
    P \left( A_2, A_1 \right) = P \left( A_2 \mid A_1 \right) \cdot P \left( A_1 \right)
  \end{equation}
\end{framed}
